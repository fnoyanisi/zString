73 zrepstr(char *str, const char *x, const char *y){
74  char *p = str;
75  size_t tlen = strlen(str); /* total length of the buffer */
76  size_t clen, ylen = strlen(y); /* clen - copy length */
77 
78  while ((p = strstr(str, x))!= NULL) {
79  /* litte bit of pointer arithmetic
80  *
81  * tlen (str + tlen)
82  * |-----------------------|
83  * | | | |
84  * str................p.............(p + ylen)
85  * |<--------------->|<---->
86  * (p - str) (str + tlen - p)
87  */
88  clen = ((p - str + ylen) <= tlen) ? ylen : (str + tlen - p);
89  p = strncpy(p, y, clen);
90  p += clen;
91  }
92  return str;
93 }
